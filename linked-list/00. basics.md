# Linked list
- Data structure used to store the similar type of data.
- collection of node that contain data and address of the next node.
- node contain data and the pointer pointing to next data.

### If array is already there, then why we need this?
- In array, reversing or adding/removing from in-between/starting is not efficient
- elements size is independent of other elements(not in contiguos fashion).


## Creation

````js
class Node {
    constructor(data) {
        this.data = data;
        this.next = null;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
    }

    addLast(data) {
        const node = new Node(data);
        if (!this.head) {
            this.head = node;
        } else {
            let current = this.head;
            while (current.next) {
                current = current.next;
            }
            current.next = node;
        }
    }

    addFirst(data) {
        let node = new Node(data);
        if (!this.head) {
            this.head = node;
        } else {
            node.next = this.head;
            this.head = node;
        }
    }

    print() {
        let current = this.head
        while (current) {
            console.log(current)
            current = current.next
        }
    }

}

const ll = new LinkedList();
ll.addLast(10);
ll.addLast(15);
ll.addLast(20);

ll.addFirst(2);
ll.addFirst(5);

ll.print()


function insertInLL(arr) {
    const newLL = new LinkedList();
    for (let i = 0; i < arr.length; i++) {
        newLL.addLast(arr[i]);
    }
    return newLL.print();
}

insertInLL([2, 4, 6, 9]);
````
